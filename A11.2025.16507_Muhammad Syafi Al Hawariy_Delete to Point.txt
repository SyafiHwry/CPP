#include <iostream>
using namespace std;

struct Node{
    int data;
    Node* next;
};  

int search(Node* head, int value) {
    Node* current = head;
    while (current != nullptr) {
        if (current->data == value) {
            cout << "Data " << value << " ditemukan" << endl;
            return value;
        }
        current = current->next;
    }
    cout << "Data tidak ditemukan" << endl;
    return -1;
}

void deletePoint(Node* head, int value) {
    Node* prev = head;
    while (prev->next != nullptr && prev->next->data != value) {
        prev = prev->next;
    }
    if (prev->next != nullptr) {
        Node* target = prev->next;
        prev->next = target->next;
        delete target;
    } else {
        cout << "Data tidak ditemukan" << endl;
    }
}

void printList(Node* head) {
    cout << "Linked list: ";
    Node* current = head;
    while(current != nullptr) {
        cout << current->data <<" ";
        current = current->next;
    }
    
}

int main() {
    int value;
    Node* head = new Node;
    head->data = 111;
    head->next = new Node;
    head->next->data = 2025;
    head->next->next = new Node;
    head->next->next->data = 16507;
    head->next->next->next = nullptr;

    cout << "Linked List: ";
    printList(head);
    cout <<endl;

    cout << "Data yang ingin anda hapus: ";
    cin >> value;
    search(head, value);
    if (head != nullptr && head->data == value) {

        }
    } else {
        deletePoint(head, value);
    }

    cout << "Data setelah terhapus: ";
    printList(head);
}
